1、请从表中查询工作是CLERK的所有人的姓名、工资、部门号、部门名称以及部门地址的信息
SELECT EMP.ENAME, EMP.SAL, EMP.DEPTNO, DEPT.DNAME, DEPT.LOC
FROM EMP
JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.JOB = 'CLERK';

2、查询表EMP中所有的工资大于等于2000的雇员姓名和他的经理的名字
SELECT E1.ENAME AS EMPLOYEE_NAME, E2.ENAME AS MANAGER_NAME
FROM EMP E1
JOIN EMP E2 ON E1.MGR = E2.EMPNO
WHERE E1.SAL >= 2000;

3、在表EMP中查询所有工资高于JONES的所有雇员姓名、工作和工资
SELECT ENAME, JOB, SAL
FROM EMP
WHERE SAL > (SELECT SAL FROM EMP WHERE ENAME = 'JONES');

4、列出没有对应部门表信息的所有雇员的姓名、工作以及部门号
SELECT EMP.ENAME, EMP.JOB, EMP.DEPTNO
FROM EMP
LEFT JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO
WHERE DEPT.DEPTNO IS NULL;

5、查找工资在1000～3000之间的雇员所在部门的所有人员信息
SELECT EMP.ENAME, EMP.SAL, DEPT.DEPTNO, DEPT.DNAME, DEPT.LOC
FROM EMP
JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.SAL BETWEEN 1000 AND 3000;

6、查询所有雇员的姓名、SAL与COMM之和
SELECT ENAME, SAL, COMM, (SAL + COALESCE(COMM, 0)) AS TOTAL_INCOME
FROM EMP;

7、查询所有81年7月1日以前来的员工姓名、工资、所属部门的名字
SELECT EMP.ENAME, EMP.SAL, DEPT.DNAME
FROM EMP
JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.HIREDATE < '1981-07-01';

8、查询各部门中81年1月1日以后来的员工数
SELECT EMP.ENAME, EMP.SAL, DEPT.DNAME
FROM EMP
JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO
WHERE EMP.HIREDATE < '1981-07-01';

9、查询所有在CHICAGO工作的经理MANAGER和销售员SALESMAN的姓名、工资
SELECT ENAME, JOB, SAL
FROM EMP
WHERE JOB IN ('MANAGER', 'SALESMAN') AND DEPTNO IN (SELECT DEPTNO FROM DEPT WHERE LOC = 'CHICAGO');

10、查询于81年来公司所有员工的总收入（SAL和COMM）
SELECT SUM(SAL + COALESCE(COMM, 0)) AS TOTAL_INCOME
FROM EMP
WHERE EXTRACT(YEAR FROM HIREDATE) = 1981;

11、查询公司中按年份月份统计各地的录用职工数量
SELECT EXTRACT(YEAR FROM HIREDATE) AS YEAR, EXTRACT(MONTH FROM HIREDATE) AS MONTH, DEPT.LOC, COUNT(*) AS NUM_OF_EMPLOYEES
FROM EMP
JOIN DEPT ON EMP.DEPTNO = DEPT.DEPTNO
GROUP BY EXTRACT(YEAR FROM HIREDATE), EXTRACT(MONTH FROM HIREDATE), DEPT.LOC;

12、查询列出各部门的部门名和部门经理名字
SELECT DEPT.DNAME, EMP.ENAME AS MANAGER_NAME
FROM DEPT
JOIN EMP ON DEPT.MGR = EMP.EMPNO;

13、 查询部门平均工资最高的部门名称
SELECT DEPT.DNAME
FROM DEPT
JOIN EMP ON DEPT.DEPTNO = EMP.DEPTNO
GROUP BY DEPT.DNAME
ORDER BY AVG(EMP.SAL) DESC
LIMIT 1;

14、从emp中查出雇员编号为7369的员工的信息(显示雇员编号，雇员名称，从事工作，工资)
SELECT EMPNO, ENAME, JOB, SAL
FROM EMP
WHERE EMPNO = 7369;

16、从emp中找出工资为1500到4000之间的员工，并且从事的是销售的工作

17、从emp中找出工资为1500到4000之间的员工，按工资的降序排列，工资相同的按入职日期降序排列

FROM EMP

27、列出在部门“SALES”工作的员工的姓名，并按工资降序排列
SELECT ENAME
FROM EMP
WHERE DEPTNO = (SELECT DEPTNO FROM DEPT WHERE DNAME = 'SALES')

28、列出薪金高于在部门30工作的所有的员工的薪金的员工姓名和薪金
SELECT E1.ENAME, E1.SAL
FROM EMP E1
JOIN EMP E2 ON E1.DEPTNO = E2.DEPTNO
WHERE E2.DEPTNO = 30 AND E1.SAL > E2.SAL;
